public class Main {

    static final int MAX = 1550;

    static int n, items[]; // ì§€ì—­ì˜ ìˆ˜, ê° ì§€ì—­ì˜ ì•„ì´í…œ ìˆ˜
    static int[][] dist; // ê° ì§€ì—­ ê°„ì˜ ê±°ë¦¬

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();

        // ì…ë ¥ ë° ì „ì²˜ë¦¬
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken()); // m: ìˆ˜ìƒ‰ ë²”ìœ„
        int r = Integer.parseInt(st.nextToken()); // r: ê¸¸ì˜ ê°œìˆ˜

        st = new StringTokenizer(br.readLine(), " ");
        items =new int[n + 1];
        for (int i = 1; i <= n; ++i) {
            items[i] = Integer.parseInt(st.nextToken());
        }

        dist = new int[n + 1][n + 1];
        for (int i = 1; i <= n; ++i) {
            Arrays.fill(dist[i], MAX);
            dist[i][i] = 0;
        }
        for (int i = 0; i < r; ++i) {
            st = new StringTokenizer(br.readLine(), " ");
            int from = Integer.parseInt(st.nextToken());
            int to = Integer.parseInt(st.nextToken());
            int len = Integer.parseInt(st.nextToken());

            dist[from][to] = len; dist[to][from] = len;
        }

        // ê° ì§€ì—­ ê°„ ìµœë‹¨ ê±°ë¦¬ êµ¬í•˜ê¸°
        floydwarshall();

        int max = 0;
        for (int i = 1; i <= n; ++i) {
            int sum = 0;
            for (int j = 1; j <= n; ++j) {
                if(dist[i][j] <= m) sum += items[j];
            }
            if (sum > max) max = sum;
        }
        sb.append(max).append("\n");

        br.close();

        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

    private static void floydwarshall() {
        for (int k = 1; k <= n; ++k) {
            for (int i = 1; i <= n; ++i) {
                if (k == i) continue;
                for (int j = 1; j <= n; ++j) {
                    if (k == j || i == j) continue;

                    dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
    }
}
