import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

    static int[] dr = {-1, 1, 0, 0};
    static int[] dc = {0, 0, -1, 1};

    static int r, c;
    static int[][] map;
    static Queue<int[]> jihun = new LinkedList<>();
    static Queue<int[]> fire = new LinkedList<>();

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();

        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        r = Integer.parseInt(st.nextToken());
        c = Integer.parseInt(st.nextToken());

        map = new int[r][c];
        for (int i = 0; i < r; ++i) {
            String row = br.readLine();
            for (int j = 0; j < c; ++j) {
                char stat = row.charAt(j);
                if(stat == '#') map[i][j] = -1;
                else if (stat == 'J') {
                    map[i][j] = 1;
                    jihun.offer(new int[]{i, j, 1});
                }
                else if (stat == 'F') {
                    map[i][j] = -1;
                    fire.offer(new int[]{i, j, 1});
                }
            }
        }

        int jx = jihun.peek()[0]; int jy = jihun.peek()[1];
        if (jx == 0 || jx == r - 1 || jy == 0 || jy == c - 1) sb.append("1\n");
        else {
            int step = solve();
            sb.append(step > 0 ? step : "IMPOSSIBLE").append("\n");
        }

        bw.write(sb.toString());
        bw.flush();
        bw.close();
        br.close();
    }

    static int solve() {
        int step = 0;
        while (!jihun.isEmpty()) {
            step += 1;
            fireMove(step);
            int jm = jihunMove(step);
            if (jm > 0) return jm;
        }
        return -1;
    }

    static boolean check(int x, int y) {
        if (x < 0 || x >= r || y < 0 || y >= c) return false;
        if (map[x][y] != 0) return false; // ì´ë¯¸ ë°©ë¬¸í–ˆê±°ë‚˜ ê°ˆ ìˆ˜ ì—†ëŠ” ê³³
        return true;
    }

    static void fireMove(int step) {
        while (!fire.isEmpty() && fire.peek()[2] == step) {
            int[] cur = fire.poll();

            for (int dir = 0; dir < 4; ++dir) {
                int nr = cur[0] + dr[dir];
                int nc = cur[1] + dc[dir];

                if (!check(nr, nc)) continue;

                map[nr][nc] = -1;
                fire.offer(new int[]{nr, nc , step + 1});
            }
        }
    }

    static int jihunMove(int step) {
        while (!jihun.isEmpty() && jihun.peek()[2] == step) {
            int[] cur = jihun.poll();

            for (int dir = 0; dir < 4; ++dir) {
                int nr = cur[0] + dr[dir];
                int nc = cur[1] + dc[dir];

                if (!check(nr, nc)) continue;
                if (nr == 0 || nr == r - 1 || nc == 0 || nc == c - 1) return cur[2] + 1; // ê°€ì¥ ìë¦¬ì— ë„ì°©

                map[nr][nc] = cur[2] + 1;
                jihun.offer(new int[]{nr, nc ,cur[2]+ 1});
            }
        }
        return -1;
    }
}
